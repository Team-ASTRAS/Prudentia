#Switch to Desktop directory:
cd Desktop

#Clone prudentia repository:
sudo git clone <prudentia repo>

#Set permissions:
sudo chmod u=wrx -R  Prudentia

#Enable camera:
sudo raspi-config
#Go to interface settings -> Camera -> enable -> finish -> reboot
#Make sure camera is installed correctly on boot

#Install picamera
sudo apt-get update
sudo apt-get install python-picamera python3-picamera

#Install cv2
sudo pip3 install opencv-python

#Install websockets:
sudo pip3 install websockets

#Install numpy / scipy
sudo pip3 install scipy


#Install python CRC version 1.21
sudo pip3 install pythoncrc

#Install pyvesc
sudo pip3 install pyvesc

#Install Custom PyVESC Code
#Download PyVESC and modify:
#PyVESC/pyvesc/VESC/messages/getters.py -> duty field should read 'duty_now' (Check both field lists)

sudo python3 -m pip install -e /home/pi/Desktop/test/PyVESC
#This runs setup.py and installs lib to pip3

If numpy gives the error "cannot open shared object file: No such file or directory" the most likely solution is to run "sudo apt-get install libatlas-base-dev" see https://numpy.org/devdocs/user/troubleshooting-importerror.html
